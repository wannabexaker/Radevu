### `apps/web/middleware.ts`

Pseudocode:

```ts
import { NextResponse, type NextRequest } from "next/server";
import { isReservedSlug, parseSubdomain, slugFromRequest } from "@/lib/routing";
import { env } from "@/lib/env";

export const config = {
  matcher: ["/((?!_next/|favicon.ico|api/health).*)"]
};

export function middleware(req: NextRequest) {
  const url = req.nextUrl.clone();
  const mode = env.ROUTING_MODE;
  const base = env.BOOKING_BASE_DOMAIN;
  const host = req.headers.get("host") ?? "";

  if (mode === "subdomain") {
    const sub = parseSubdomain(host, base);
    if (sub === null || sub === "" || sub === "www") {
      return NextResponse.next(); // root marketing
    }
    if (sub === "dashboard") {
      // /dashboard/* paths pass through untouched.
      return NextResponse.next();
    }
    if (isReservedSlug(sub)) {
      return new NextResponse(null, { status: 404 });
    }
    url.pathname = `/${sub}${url.pathname === "/" ? "" : url.pathname}`;
    return NextResponse.rewrite(url);
  }

  // subpath mode
  const segs = url.pathname.split("/").filter(Boolean);
  const first = segs[0];
  if (!first) return NextResponse.next();
  if (first === "dashboard" || first === "api") return NextResponse.next();
  if (isReservedSlug(first)) {
    return new NextResponse(null, { status: 404 });
  }
  // /{slug}/... → rewrites to /[business] with slug context already captured by file-based route
  return NextResponse.next();
}
```

In subpath mode, the dynamic route `/[business]/page.tsx` captures `params.business` directly — no rewrite needed. The middleware's job in subpath mode is only to 404 reserved slugs early.
